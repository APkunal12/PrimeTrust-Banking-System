pipeline {
    agent any

    stages {

        stage('Clone GitHub Repository') {
            steps {
                git 'https://github.com/APkunal12/PrimeTrust-Banking-System'
            }
        }

        stage('Build Docker Image') {
            steps {
                sh 'sudo docker build -t banking-app .'
            }
        }

        stage('Run Docker Container') {
            steps {
                sh 'sudo docker run -d -p 80:8080 banking-app'
            }
        }

    }
}
